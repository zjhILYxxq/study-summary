#### 数组扁平化

**数组扁平化**的原生方法:
```
let arr = [1, [2, [3, 4, [5]]]];

arr.flat(4);
```
**数组扁平化**的核心思想: **递归**。

```
function flat(list, deep) {
    return list.reduce((initial, item) => {
        if (Array.isArray(item) && deep) {
            return [...initial, ...flat(item, deep - 1)]
        } else {
            return [...initial, item];
        }
    }, []);
}
```

#### 扁平数据结构和 tree 相互转换

扁平化数据结构 -> tree

```
function arrayToTree(list) {
    let result = [];
    let cache = {}
    for(let item of list) {
        cache[item.id] = {
            ...item,
            children: cache[item.id] ? cache[item.id].children : []
        }
        if (!cache[item.pid]) {
            cache[item.pid] = {
                children: []
            }
        }
        if (cache[item.pid]) {
            cache[item.pid].children.push(cache[item.id]);
        }
        if (item.pid === 0) {
            result.push(cache[item.id]);
        }
    }
    return result;
}
```

function treeToArray(tree) {
    let result = [];
    if (Array.isArray(tree)) {
        tree.forEach(item => {
            travel(item, result);
        })
    } else {
        travel(tree, result);
    }
    return result;
}

function travel(tree, list) {
    if (tree) {
        list.push(tree);
        if (tree.children && tree.children.length) {
            tree.children.forEach(item => {
                travel(item, list);
            })
        }
    }
}

#### 排序

- **快速排序**

    **快速排序**的核心思想是: **递归**。
    ```
    function quickSort(nums, start, end) {
        if (start >= end) return;
        let privot = position(nums, start, end);
        quickSort(nums, start, privot - 1);
        quickSort(nums, privot + 1, end);
    }

    function swap(nums, i, j) {
        let temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
    }

    function position(nums, start, end) {
        let left = start, right = end, privotIndex = start, privot = nums[start];
        while(left < right) {
            if (nums[left] <= privot) {
                left++
            } else if (nums[right] >= privot) {
                right--
            } else {
                swap(nums, left, right);
                left++;
                right--
            }
        }
        if(privot < nums[left]) {
            swap(nums, privotIndex, left - 1);
            return left - 1;
        } else {
            swap(nums, privotIndex, left);
            return left;
        }
    }
    ```
